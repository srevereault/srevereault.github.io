L'écran télécommandé

Code source Python


import lirc
import subprocess
import time

def initir():
    sockid = lirc.init("brain", blocking = False)

def launch_epi():
    epiph = subprocess.Popen(["epiphany", "file:///home/pi/logo.html"])
    time.sleep(10)
    subprocess.Popen(["xte", "key F11"])

def loadvideo():
    subprocess.Popen(["xte", "key j"])

def reload():
    subprocess.Popen(["xte", "key r"])

def shutdown():
    subprocess.Popen(["sudo", "halt", "-p"])

initir()
launch_epi()

while True:
    codeIR = lirc.nextcode()
    if codeIR:
        if codeIR[0] == "play":
            loadvideo()
        if codeIR[0] == "reload":
            reload()
        if codeIR[0] == "stop":
            shutdown()
